#Elections of 1936
country_event = {
	id = japan.2
	title = japan.2.t
	desc = japan.2.desc
	picture = GFX_report_event_election_vote

	trigger = {
		original_tag = JAP
		has_elections = yes
		date > 1936.2.19
	}
	
	fire_only_once = yes
	
	option = { #
		name = japan.2.a
		trigger = {
			always = no
		}

#The Showa Reform Bill
country_event = {
	id = japan.5
	title = japan.5.t
	desc = japan.5.desc
	#picture = GFX_report_event_chinese_soldiers_01

	is_triggered_only = yes
	
	option = { #The Bill Passes
		name = japan.5.a
		remove_ideas = JAP_rise_of_the_masses
		add_timed_idea = {
			idea = JAP_mass_reforms
			days = 180
		}
		set_country_flag = JAP_the_bill_passed
	}
	option = { #The Bill Fails
		name = japan.5.b
		trigger = {
			always = no
		}
		swap_ideas = {
			remove_idea = JAP_rise_of_the_masses
			add_idea = JAP_rise_of_the_masses1
		}
		set_country_flag = JAP_the_bill_failed
	}
}
