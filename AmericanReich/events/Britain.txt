#ENGSOC Victory!
country_event = {
	id = britain.8
	title = britain.8.t
	desc = britain.8.desc
	#picture = GFX_report_event_physics_lab_01
	
	mean_time_to_happen = {
		days = 1
	}
	
	fire_only_once = yes
	
	trigger = {
		tag = ENG
		has_completed_focus = ENG_convenetherevolutionarycongress
		OR = {
			AND = {
				nationalist > 0.4
				syndicalist < 0.4
				autocratic < 0.4
			}
			AND = {
				nationalist < 0.4
				syndicalist < 0.4
				autocratic < 0.4
			}
		}
	}
	
	option = {
		name = britain.8.a	#Glory to Britain!
		ai_chance = {
			factor = 50 
		}
		set_politics = {
			ruling_party = nationalist
		}
		add_popularity = {
			ideology = nationalist
			popularity = 0.05
		}
		add_popularity = {
			ideology = populist
			popularity = 0.1
		}
		set_country_flag = ENG_revolutionary_congress_finished
		if = {
			limit = {
				has_idea = ENG_oswald_mosley
			}
			add_ideas = {
				ENG_cynthia_mosley
			}
		}
	}
}
